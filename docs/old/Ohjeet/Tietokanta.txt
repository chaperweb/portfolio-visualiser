=Setup=
  ==Asenna paketit==
  yum -y install postgresql postgresql-server postgresql-devel postgresql-contrib postgresql-docs
  service postgresql initdb
  chkconfig postgresql on
  service postgresql start
  
  ==Luo tietokanta ja käyttäjä==
  sudo -u postgres psql
  postgres=# create database portfolio_visualiser;
  postgres=# create user portfolio_visualiser with password 'portfolio_visualiser';
  postgres=# alter role portfolio_visualiser set client_encoding to 'utf8';
  postgres=# alter role portfolio_visualiser set default_transaction_isolation to 'read committed';
  postgres=# alter role portfolio_visualiser set timezone to 'UTC';
  postgres=# grant all privileges on database portfolio_visualiser to portfolio_visualiser;
  
  ==Lisää tiedostoon /var/lib/pgsql9/data/pg_hba.conf rivin "local   all             all" jälkeen==
  # Portfolio visualizer
  host    portfolio_visualiser portfolio_visualiser 127.0.0.1/32  md5
  
  ==Käynnistä Postgresql uudelleen==
  service postgresql restart
  
  ==Testaa luodut tunnukset==
  psql -h 127.0.0.1 -d portfolio_visualiser -U portfolio_visualiser -W
